## 快速排序

```typescript
/**
 * @param {number[]} stones
 * @return {number}
 */
var lastStoneWeightII = function (stones) {
  let sum = stones.reduce((a, b) => a + b)
  let dp = [], mid = sum / 2 | 0
  for (let val of stones) {
      for (let i = mid; i >= val; i--) {
          dp[i] = Math.max(dp[i] || 0, (dp[i - val] || 0) + val)
      }
  }
  return sum - 2 * dp[mid]
};
```
